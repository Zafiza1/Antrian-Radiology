    <script>
        function escapeHtml(text) {
            if (text === null || text === undefined) return '';
            const div = document.createElement('div');
            div.textContent = String(text);
            return div.innerHTML;
        }

        const pusher = new Pusher('{{ config('broadcasting.connections.pusher.key') }}', {
            cluster: '{{ config('broadcasting.connections.pusher.options.cluster') }}',
            forceTLS: {{ config('broadcasting.connections.pusher.options.useTLS') ? 'true' : 'true' }}
        });

        const channel = pusher.subscribe('patient-calls');

        function updateInfoNext(next) {
            const infoBox = document.querySelector('.info-next');
            if (!infoBox) return;

            if (next) {
                infoBox.innerHTML = `
                    <p class="text-xl text-gray-700">
                        Pasien selanjutnya yang akan dipanggil:
                        <span class="font-bold text-blue-600">${escapeHtml(next.patient_name)}</span>
                    </p>
                    <p class="text-gray-600 mt-2">
                        Pemeriksaan: ${escapeHtml(next.examination_name)}
                    </p>
                `;
            } else {
                infoBox.innerHTML = `
                    <p class="text-xl text-gray-600">Tidak ada antrian pasien berikutnya</p>
                `;
            }
        }

        channel.bind('patient.called', function(data) {
            updatePatientDisplay(data.current, data.next);
            updateInfoNext(data.next);

            if (data.current && data.current.patient_name) {
                speak(`Panggilan kepada pasien atas nama ${data.current.patient_name} yang sedang dipanggil agar menuju ke Ruang Radiologi`);
            }
        });

        function updatePatientDisplay(current, next) {
            const currentPatientDiv = document.querySelector('.current-patient');
            if (current) {
                currentPatientDiv.innerHTML = `
                    <div class="text-4xl font-bold text-blue-600 mb-2">
                        ${escapeHtml(current.patient_name)}
                    </div>
                    <div class="text-lg text-gray-600">
                        Pemeriksaan: ${escapeHtml(current.examination_name)}
                    </div>
                    <div class="text-gray-500 mt-2">
                        ${new Date(current.arrival_date).toLocaleTimeString('id-ID', {hour: '2-digit', minute:'2-digit', hour12: false})} WIB
                    </div>
                `;
            } else {
                currentPatientDiv.innerHTML = `
                    <div class="text-4xl font-bold text-gray-400 py-8">-</div>
                    <p class="text-gray-500">Tidak ada pasien yang sedang dipanggil</p>
                `;
            }

            const nextPatientDiv = document.querySelector('.next-patient');
            if (next) {
                nextPatientDiv.innerHTML = `
                    <div class="text-4xl font-bold text-yellow-500 mb-2">
                        ${escapeHtml(next.patient_name)}
                    </div>
                    <div class="text-lg text-gray-600">
                        Pemeriksaan: ${escapeHtml(next.examination_name)}
                    </div>
                    <div class="text-gray-500 mt-2">
                        ${new Date(next.arrival_date).toLocaleTimeString('id-ID', {hour: '2-digit', minute:'2-digit', hour12: false})} WIB
                    </div>
                `;
            } else {
                nextPatientDiv.innerHTML = `
                    <div class="text-4xl font-bold text-gray-400 py-8">-</div>
                    <p class="text-gray-500">Tidak ada antrian berikutnya</p>
                `;
            }
        }

        function speak(text) {
            if ('speechSynthesis' in window) {
                try {
                    window.speechSynthesis.cancel();
                } catch (e) {}

                const speech = new SpeechSynthesisUtterance();
                speech.text = text;
                speech.lang = 'id-ID';
                window.speechSynthesis.speak(speech);
            }
        }
    </script>
